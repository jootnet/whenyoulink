温油轻远程 import { $, on } from "@sciter"; const window = Window.this const closeBtn = $('#closeBtn') const minimizeBtn = $('#minimizeBtn') closeBtn.on('mouseenter', () => { closeBtn.firstElementChild.classList.add("hover") }) closeBtn.on('mouseleave', () => { closeBtn.firstElementChild.classList.remove("hover") }) on('click', '#closeBtn', () => { window.close(true) }) on('click', '#minimizeBtn', () => { window.state = Window.WINDOW_MINIMIZED }) document.on('ready', () => { globalThis.peerId = window.parameters.peerId globalThis.webrtc = window.parameters.webrtc globalThis.webrtc.genSDP() }) document.on('beforeunload', () => { window.parent.postEvent(new Event('childClose', {data: {peerId: globalThis.peerId}})) }) window.on('signalMsgReceived', (evt) => { })
